/*客户信息表*/
DROP TABLE IF EXISTS `systemcustomer`;
CREATE TABLE `systemcustomer` (
  `AutoID` INT(10) NOT NULL AUTO_INCREMENT COMMENT '客户ID',
  `Title` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '客户名称',  
  `Status` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '客户状态-1：有效，0：无效',
  `LinkMan` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '联系人',  
  `Phone` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '联系电话',  
  `MobilePhone` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '移动电话',  
  `QQ` VARCHAR(50) NOT NULL DEFAULT '' COMMENT 'QQ',  
  `Fax` VARCHAR(50) NOT NULL DEFAULT '' COMMENT 'Fax', 
  `Email` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '邮箱',   
  `Address` VARCHAR(50) NOT NULL DEFAULT '' COMMENT '地址',  
  `OrderValue` INT(10) NOT NULL  DEFAULT 0 COMMENT '排序值',
  `Remark` VARCHAR(4000) NOT NULL  DEFAULT '' COMMENT '备注',
  `CreatorID` INT(10) NOT NULL DEFAULT 0 COMMENT '创建人id',
  `CreateDate` DATETIME  NOT NULL  DEFAULT '2012-8-16 00:00:00' COMMENT '创建时间',
  `Modifier` INT(10) NOT NULL DEFAULT 0 COMMENT '修改人id',
  `ModifyDate` DATETIME  NOT NULL   DEFAULT '2012-8-16 00:00:00'   COMMENT '修改时间',
  PRIMARY KEY (`AutoID`),
  UNIQUE KEY `PK_systemcustomer` (`AutoID`)
) ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;
