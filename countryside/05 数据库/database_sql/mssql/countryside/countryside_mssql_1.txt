/*客户信息表*/
if exists (select top 1 * from sysobjects where id = object_id(N'systemcustomer') and type='U')
drop table systemcustomer
 
CREATE TABLE systemcustomer (
  AutoID INT  NOT NULL   IDENTITY(1,1)  primary key,---COMMENT '客户ID'
  Title VARCHAR(50) NOT NULL DEFAULT '',---COMMENT '客户名称' 
  Status TINYINT NOT NULL DEFAULT 0,  --COMMENT '客户状态-1：有效，0：无效',
  LinkMan VARCHAR(50) NOT NULL DEFAULT '', --COMMENT '联系人',  
  Phone VARCHAR(50) NOT NULL DEFAULT '', --COMMENT '联系电话',  
  MobilePhone VARCHAR(50) NOT NULL DEFAULT '' ,--COMMENT '移动电话',  
  QQ VARCHAR(50) NOT NULL DEFAULT '', --COMMENT 'QQ',  
  Fax VARCHAR(50) NOT NULL DEFAULT '' ,--COMMENT 'Fax',
  Email VARCHAR(50) NOT NULL DEFAULT '' ,--COMMENT '邮箱',  
  Address VARCHAR(50) NOT NULL DEFAULT '', --COMMENT '地址',  
  OrderValue INT NOT NULL  DEFAULT 0 ,--COMMENT '排序值',
  ProvinceID INT NOT NULL  DEFAULT 0 ,--COMMENT '所属省',
  CityID INT NOT NULL  DEFAULT 0 ,--COMMENT '所属市',
  CountyID INT NOT NULL  DEFAULT 0 ,--COMMENT '所属县',
  TownID INT NOT NULL  DEFAULT 0 ,--COMMENT '所属镇',
  Remark VARCHAR(4000) NOT NULL  DEFAULT '', --COMMENT '备注',
  CreatorID INT  NOT NULL DEFAULT 0, --COMMENT '创建人id',
  CreateDate DATETIME  NOT NULL  DEFAULT '2012-8-16 00:00:00', --COMMENT '创建时间',
  Modifier INT  NOT NULL DEFAULT 0, --COMMENT '修改人id',
  ModifyDate DATETIME  NOT NULL   DEFAULT '2012-8-16 00:00:00'   --COMMENT '修改时间',
)  
GO

CREATE TABLE systemmenu (
  AutoID INT NOT NULL identity(1,1) primary key,--COMMENT '系统菜单ID',
  CustomerID INT  NOT NULL DEFAULT 0,-- COMMENT '客户ID',
  Title VARCHAR(50) NOT NULL DEFAULT '' ,--COMMENT '菜单名称',  
  MenuUrl VARCHAR(50) NOT NULL DEFAULT '' ,--COMMENT '菜单Url', 
  Alt  VARCHAR(200) NOT NULL DEFAULT '' ,--COMMENT '菜单描述', 
  IsSuperAdmin   TINYINT  NOT NULL --DEFAULT 0 COMMENT '是否专属超级管理员',
  IsTopMenu INT  NOT NULL DEFAULT 0 ,--COMMENT '是否顶级菜单，1：是， 0：不是', 
  TopMenuID INT  NOT NULL DEFAULT 0 ,--COMMENT '所属的顶级菜单ID,二级和三级菜单所属顶级菜单ID,如是顶级菜单显示0', 
  ParentId INT  NOT NULL DEFAULT 0 ,--COMMENT '所属的上级菜单ID，顶级菜单和二级菜单都为0',    
  MenuType VARCHAR(1)  NOT NULL DEFAULT '' ,--COMMENT '菜单类型:0:主页面,2：分页面,1:分页面（必须要新增 修改 删除） ',    
  Status TINYINT  NOT NULL DEFAULT 0 ,--COMMENT '菜单状态-1：有效，0：无效',
  OrderValue INT  NOT NULL  DEFAULT 0 ,--COMMENT '排序值',
  Remark VARCHAR(4000) NOT NULL  DEFAULT '',-- COMMENT '备注',
  CreatorID INT  NOT NULL DEFAULT 0 ,--COMMENT '创建人id',
  CreateDate DATETIME  NOT NULL  DEFAULT '2012-8-16 00:00:00' ,--COMMENT '创建时间',
  Modifier INT  NOT NULL DEFAULT 0 ,--COMMENT '修改人id',
